          <div key={idx} className="bg-white border-l-4 border-orange-500 rounded-lg p-4 shadow">
            <h3 className="font-bold text-orange-700 text-lg mb-2">{rule.rule}</h3>
            <p className="text-gray-700 text-sm mb-3">{rule.detail}</p>
            <p className="text-red-700 font-semibold text-sm">{rule.danger}</p>
          </div>
        ))}
      </div>

      <div className="bg-red-100 border-l-4 border-red-600 p-6 rounded-lg mt-6">
        <h2 className="font-bold text-red-900 text-lg mb-3">🚨 CONTRAFLOW SURVIVAL TIPS</h2>
        <ul className="text-red-900 space-y-2">
          <li>✓ Reduce speed to 40-50 mph maximum</li>
          <li>✓ Do NOT change lanes unless absolutely necessary</li>
          <li>✓ Never cross the center barrier or line</li>
          <li>✓ Stay in your designated lane at all times</li>
          <li>✓ Watch other vehicles - they may panic or swerve</li>
          <li>✓ Use dipped headlights even in daytime</li>
          <li>✓ Maintain 3+ second following distance</li>
          <li>✓ No distractions - phones away, focus on road</li>
          <li>✓ If tired, EXIT and take a break</li>
          <li>✓ Never undertake (pass on left)</li>
        </ul>
      </div>

      <div className="bg-yellow-50 border-l-4 border-yellow-600 p-4 rounded-lg">
        <p className="text-yellow-900"><strong>📊 Statistics:</strong> Contraflow sections have higher accident rates. Many involve fatalities. The main cause is lane changing and driver distraction. One wrong move can result in a head-on collision.</p>
      </div>
    </div>
  );

  const renderEmergency = () => (
    <div className="space-y-4">
      <h2 className="text-3xl font-bold text-blue-900 mb-6">🚨 Emergency Procedures</h2>
      <p className="text-gray-600 mb-6">Know what to do in critical situations. Your actions can save lives.</p>
      
      <div className="space-y-4">
        {emergencyRules.map((emergency, idx) => (
          <div key={idx} className="bg-white border-l-4 border-red-600 rounded-lg overflow-hidden shadow">
            <button
              onClick={() => toggleExpanded(`emergency-${idx}`)}
              className="w-full p-4 bg-red-50 hover:bg-red-100 flex items-center justify-between"
            >
              <h3 className="font-bold text-red-700">{emergency.situation}</h3>
              <ChevronDown 
                size={20} 
                className={`text-red-600 transition-transform ${expandedItems[`emergency-${idx}`] ? 'rotate-180' : ''}`}
              />
            </button>
            {expandedItems[`emergency-${idx}`] && (
              <div className="p-4 bg-white space-y-3">
                <div className="bg-yellow-50 p-3 rounded border-l-2 border-yellow-500">
                  <p className="text-yellow-900 font-bold">{emergency.critical}</p>
                </div>
                <div>
                  <p className="font-bold text-gray-800 mb-2">Steps to Follow:</p>
                  <ol className="space-y-1">
                    {emergency.steps.map((step, sidx) => (
                      <li key={sidx} className="text-gray-700 text-sm">{step}</li>
                    ))}
                  </ol>
                </div>
              </div>
            )}
          </div>
        ))}
      </div>

      <div className="grid md:grid-cols-2 gap-4 mt-6">
        <div className="bg-blue-50 border-l-4 border-blue-600 p-4 rounded">
          <h3 className="font-bold text-blue-700 mb-2">📞 Emergency Numbers</h3>
          <ul className="text-sm text-blue-900 space-y-1">
            <li><strong>999</strong> - Police, Fire, Ambulance</li>
            <li><strong>112</strong> - EU Emergency (EU roaming)</li>
            <li><strong>101</strong> - Non-emergency Police</li>
            <li><strong>101</strong> - Breakdown services</li>
          </ul>
        </div>

        <div className="bg-green-50 border-l-4 border-green-600 p-4 rounded">
          <h3 className="font-bold text-green-700 mb-2">✓ What to Do First</h3>
          <ul className="text-sm text-green-900 space-y-1">
            <li>1. Hazard lights ON</li>
            <li>2. Move to safety if possible</li>
            <li>3. Call 999 if emergency</li>
            <li>4. Set warning triangles</li>
            <li>5. Do NOT block traffic</li>
          </ul>
        </div>
      </div>
    </div>
  );

  const renderWarnings = () => (
    <div className="space-y-4">
      <h2 className="text-3xl font-bold text-blue-900 mb-6">⚠️ UK Warning Signs</h2>
      <p className="text-gray-600 mb-6">Red triangular signs indicating hazards. Always reduce speed and be alert.</p>
      
      <div className="grid gap-4">
        {warningSignsData.map((sign, idx) => (
          <div key={idx} className="bg-white border-l-4 border-red-500 rounded-lg p-4 shadow">
            <h3 className="font-bold text-red-700">{sign.name}</h3>
            <p className="text-gray-700 mt-2"><strong>Meaning:</strong> {sign.desc}</p>
            <p className="text-gray-700"><strong>Action:</strong> {sign.action}</p>
          </div>
        ))}
      </div>
    </div>
  );

  const renderSpeedLimits = () => (
    <div className="space-y-4">
      <h2 className="text-3xl font-bold text-blue-900 mb-6">🚗 Speed Limits Overview</h2>
      <p className="text-gray-600 mb-6">National speed limits for different vehicle types and road conditions.</p>
      
      <div className="overflow-x-auto bg-white rounded-lg shadow">
        <table className="w-full">
          <thead>
            <tr className="bg-blue-700 text-white">
              <th className="px-6 py-3 text-left">Vehicle Type</th>
              <th className="px-6 py-3 text-center">Built-up</th>
              <th className="px-6 py-3 text-center">Single</th>
              <th className="px-6 py-3 text-center">Dual</th>
              <th className="px-6 py-3 text-center">Motorway</th>
            </tr>
          </thead>
          <tbody>
            {speedLimitsData.map((row, idx) => (
              <tr key={idx} className={idx % 2 === 0 ? 'bg-blue-50' : 'bg-white'}>
                <td className="px-6 py-3 font-semibold text-gray-800">{row.vehicle}</td>
                <td className="px-6 py-3 text-center text-red-600 font-bold">{row.builtup}</td>
                <td className="px-6 py-3 text-center text-red-600 font-bold">{row.single}</td>
                <td className="px-6 py-3 text-center text-red-600 font-bold">{row.dual}</td>
                <td className="px-6 py-3 text-center text-red-600 font-bold">{row.motorway}</td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>
    </div>
  );

  const renderTowing = () => (
    <div className="space-y-4">
      <h2 className="text-3xl font-bold text-blue-900 mb-6">🚙 Towing Guide</h2>
      <p className="text-gray-600 mb-6">Speed limits and requirements when towing trailers and caravans.</p>
      
      <div className="space-y-3">
        {[
          { type: 'Small Trailer (<750kg)', speeds: '30/60/70/70', req: 'No special requirements' },
          { type: 'Standard Caravan', speeds: '30/50/60/60', req: 'Brakes required, coupling certified' },
          { type: 'Large Caravan', speeds: '30/50/60/60', req: 'Servo brakes, side markers' },
        ].map((tow, idx) => (
          <div key={idx} className="bg-white border-l-4 border-green-500 rounded p-4 shadow">
            <h3 className="font-bold text-gray-800">{tow.type}</h3>
            <p className="text-sm text-gray-600 mt-2">Built-up/Single/Dual/Motorway: {tow.speeds} mph</p>
            <p className="text-sm text-green-700 font-semibold mt-1">{tow.req}</p>
          </div>
        ))}
      </div>
    </div>
  );

  const renderSection = () => {
    switch(activeSection) {
      case 'home': return renderHome();
      case 'warnings': return renderWarnings();
      case 'speedlimits': return renderSpeedLimits();
      case 'overtaking': return renderOvertaking();
      case 'parking': return renderParking();
      case 'roadstudmark': return renderRoadStudsMarks();
      case 'contraflow': return renderContraflow();
      case 'emergency': return renderEmergency();
      case 'towing': return renderTowing();
      default: return renderHome();
    }
  };

  return (
    <div className="flex h-screen bg-gray-50">
      {/* Sidebar */}
      <div className={`${mobileMenuOpen ? 'block' : 'hidden'} md:block fixed md:relative z-40 w-64 bg-blue-900 text-white shadow-lg h-screen overflow-y-auto`}>
        <div className="p-6 border-b border-blue-800">
          <h2 className="text-2xl font-bold">UK Road Rules</h2>
          <p className="text-blue-200 text-sm mt-1">Complete Guide</p>
        </div>
        
        <nav className="p-4">
          {navigationItems.map(item => (
            <button
              key={item.id}
              onClick={() => {
                setActiveSection(item.id);
                setMobileMenuOpen(false);
              }}
              className={`w-full text-left px-4 py-3 rounded-lg mb-2 transition-colors ${
                activeSection === item.id 
                  ? 'bg-blue-700 font-semibold' 
                  : 'hover:bg-blue-800'
              }`}
            >
              <span className="mr-2">{item.icon}</span>
              {item.label}
            </button>
          ))}
        </nav>

        <div className="p-4 border-t border-blue-800">
          <div className="bg-red-900 rounded-lg p-3">
            <p className="text-red-100 text-xs font-bold">⚠️ Safety First: Always prioritize safety over speed. Follow all rules and signs.</p>
          </div>
        </div>
      </div>

      {/* Main Content */}
      <div className="flex-1 flex flex-col overflow-hidden">
        {/* Top Bar */}
        <div className="bg-white border-b border-gray-200 px-6 py-4 flex items-center justify-between shadow">
          <button
            onClick={() => setMobileMenuOpen(!mobileMenuOpen)}
            className="md:hidden p-2 hover:bg-gray-100 rounded"
          >
            {mobileMenuOpen ? <X size={24} /> : <Menu size={24} />}
          </button>
          
          <div className="flex-1 mx-4">
            <h1 className="text-xl font-bold text-blue-900 hidden md:block">
              {navigationItems.find(item => item.id === activeSection)?.label || 'UK Road Rules'}
            </h1>
          </div>

          <div className="flex items-center space-x-4">
            <span className="text-sm text-gray-600 hidden sm:inline">UK Driver Guide</span>
          </div>
        </div>

        {/* Scrollable Content Area */}
        <div className="flex-1 overflow-y-auto">
          <div className="max-w-6xl mx-auto p-6">
            {renderSection()}
          </div>
        </div>

        {/* Footer */}
        <div className="bg-blue-900 text-blue-100 text-center py-4 text-xs border-t border-blue-800">
          <p>UK Road Rules & Traffic Guide © 2026 | Safety is paramount | Always obey posted signs and check official sources</p>
        </div>
      </div>
    </div>
  );
};

export default UKRoadSignsWebsite;
